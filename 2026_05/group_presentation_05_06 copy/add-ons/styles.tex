% --- Helpful macros -----------------------------------------
\newcommand{\highlight}[1]{\textcolor{Accent}{\textbf{#1}}}
\newcommand{\muted}[1]{\textcolor{Muted}{#1}}

\newenvironment{bigidea}[1]{%
  \begin{tcolorbox}[
    enhanced,
    colback=white,
    colframe=Accent,
    boxrule=0.7pt,
    arc=3mm,
    left=3mm,right=3mm,top=2mm,bottom=2mm,
    title=\textbf{#1},
    coltitle=DeepNavy,
    colbacktitle=Line,
    attach boxed title to top left={yshift=-1.5mm,xshift=3mm},
    boxed title style={arc=2mm,boxrule=0pt}
  ]
}{\end{tcolorbox}}

  \end{columns}
}
