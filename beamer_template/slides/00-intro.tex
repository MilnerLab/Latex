\section{Motivation}

\begin{frame}{A more visual opening}{Use one clear idea per slide}
  \begin{bigidea}{Main message}
    A presentation feels better when every slide has one visual hierarchy: title, key statement, then supporting detail.
  \end{bigidea}

  \vspace{0.35cm}
  \twocol{
    \textbf{Avoid}
    \begin{itemize}
      \item too many bullets
      \item default Beamer colors
      \item dense equations without context
    \end{itemize}
  }{
    \textbf{Prefer}
    \begin{itemize}
      \item large key statements
      \item restrained color accents
      \item figures with short captions
    \end{itemize}
  }
\end{frame}
