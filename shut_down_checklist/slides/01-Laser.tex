\section{Basics}

\begin{frame}{Turn off the LASER}
  \begin{enumerate}
    \item Block the laser
    \begin{itemize}
      \item block the probe and measure max compressed power
      \item block centrifuge beam before the chamber and before the centrifuge 
    \end{itemize}
    \item put optical table covers back on
    \item Turn of the laser
    \begin{enumerate}
      \item turn of \textbf{Regen} switch
      \item turn of both \textbf{ASCENT} lasers
      \item turn of \textbf{MaiTai}
      \item turn of high voltage supply via key
    \end{enumerate}
  \end{enumerate}
\end{frame}
